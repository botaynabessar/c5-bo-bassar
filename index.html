Méthodes : querySelectorAll, boucle for
Item 1
Item 2

solution.js
function colorierTous() {
  document.querySelectorAll('.item').forEach(el => el.classList.add('fond-jaune'));
}
function numeroteTous() {
  const items = document.querySelectorAll('.item');
  for (let i = 0; i < items.length; i++) {
    items[i].innerText = (i + 1) + '. Item ' + (i + 1);
  }
}
06

Sélecteur de couleur de fond

Objectif : Changer le fond d'une zone en cliquant sur une boîte.
Méthodes : style, classList, querySelectorAll


solution.js
function changerFond(zoneId, couleur, boiteCliquee) {
  document.getElementById(zoneId).style.backgroundColor = couleur;
  document.querySelectorAll('.boite').forEach(b => b.classList.remove('actif'));
  boiteCliquee.classList.add('actif');
}
function reinitialiserFond(zoneId) {
  document.getElementById(zoneId).style.backgroundColor = '';
  document.querySelectorAll('.boite').forEach(b => b.classList.remove('actif'));
}

Formulaire dynamique

Objectif : Afficher le nom en temps réel et appliquer une couleur.
Méthodes : value, innerText, style.color

Votre nom apparaîtra ici

solution.js
function afficherNom() {
  const nom = document.querySelector('#nom').value;
  document.querySelector('#affichage7').innerText =
    nom.trim() === '' ? 'Votre nom apparaîtra ici' : `Bonjour ${nom} !`;
}
function appliquerCouleur() {
  const couleur = document.querySelector('#couleur7').value;
  document.querySelector('#affichage7').style.color = couleur;
}
08

Liste TODO

Objectif : Ajouter et supprimer des tâches.
Méthodes : createElement, appendChild, remove()
    solution.js
    function ajouterTache() {
  const input = document.querySelector('#tache');
  const texte = input.value.trim();
  if (texte === '') return;
  const li = document.createElement('li');
  li.innerHTML = `<span>${texte}</span>
    <button onclick="supprimerTache(this)">X</button>`;
  document.querySelector('#listeTaches').appendChild(li);
  input.value = '';
}
function supprimerTache(b) { b.parentElement.remove(); }
function effacerTout() { document.querySelector('#listeTaches').innerHTML = ''; }

    Compteur avec conditions de couleur

    Objectif : Compteur qui change de couleur selon la valeur.
    Méthodes : innerText, classList, conditions

    0

    🔴 négatif  |  ⚫ zéro  |  🟢 positif

    solution.js
    let valeur = 0;
function incrementer() { valeur++; afficherCompteur(); }
function decrementer() { valeur--; afficherCompteur(); }
function reinitialiser() { valeur = 0; afficherCompteur(); }
function afficherCompteur() {
  const el = document.querySelector('#compteur');
  el.innerText = valeur;
  el.classList.remove('rouge', 'vert');
  if (valeur < 0) el.classList.add('rouge');
  else if (valeur > 0) el.classList.add('vert');
}

    Système d'onglets

    Objectif : Afficher un seul onglet à la fois et marquer le bouton actif.
    Méthodes : querySelectorAll, classList, forEach

    Contenu de l'onglet 1

    Lorem ipsum dolor sit amet.

    Contenu de l'onglet 2

    Consectetur adipiscing elit.

    Contenu de l'onglet 3

    Sed do eiusmod tempor.

    solution.js
    function afficherOnglet(numero) {
  document.querySelectorAll('.contenu-onglet').forEach(d => {
    d.classList.remove('visible'); d.classList.add('cache');
  });
  document.querySelector(`#contenu${numero}`).classList.replace('cache', 'visible');
  document.querySelectorAll('.onglet').forEach(b => b.classList.remove('actif'));
  document.querySelectorAll('.onglet')[numero - 1].classList.add('actif');
}
